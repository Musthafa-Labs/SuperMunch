Pincode Check (Optional for Delivery)
const serviceablePincodes = [600083, 600034, 600079, 600021];

function checkPincode(dishName){
  const pin = prompt("Enter your Pincode to check delivery:");
  if(serviceablePincodes.includes(parseInt(pin))){
    window.open(`https://wa.me/918939436427?text=Hi%20Super%20Munch,%20I%20would%20like%20to%20order%20${dishName}`,"_blank");
  } else {
    alert("Sorry! We don’t deliver to this area yet 😢");
  }
}

Works when user clicks “Order via WhatsApp”
